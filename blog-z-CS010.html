Syntax Error on Nonlocal Statement in Python2.x

  • Tomoya Machida
  • Coding

When you use 'nonlocal' statment in Pyhton3, you don't face any problem, but in Python2.x, you get Syntax Error. Because 'nonlocal' is only for python3 and pyhon2.x doesn't recognize it as it is expected. Instead, it's recognized as just letter such as 'a', 'name', 'number'.

class Solution:
    def diameterOfBinaryTree(self, root: Optional[TreeNode]) -> int:
        ans=0
        def dia(root):
            nonlocal ans
            if not root:
                return 0
            left=dia(root.left)
            right=dia(root.right)
            ans=max(ans,left+right)
            return 1+max(right,left)
        dia(root)
        return ans

class Solution(object):
    def diameterOfBinaryTree(self, root):
        dic={0:0}
        def recursion(root):
            if not root:
                return 0
            left = recursion(root.left)
            right = recursion(root.right)
            dic[0]=max(dic[0],left+right)
            return 1+max(left,right)
        recursion(root)
        return dic[0]
